let fibonacci(n : [0,]) =
    ( 0 ) when n == 0,
    ( 1 ) when n <= 2,
    default ( fibonacci(n - 1) + fibonacci(n - 2) )

let neg(n) = 0 - n

let abs(n) = ( neg(n) ) when n < 0, default ( n )

let sum(list, i : [0,]) =
    ( sum(list, size(list) - 1) ) when i == None,
    ( list[0] ) when i == 0,
    default ( sum(list, i - 1) + list[i] )
